using System.Collections;
using System.Collections.Generic;
using UnityEngine;

public class L2MoveNPCOLD : MonoBehaviour {

    private Transform playerTransform;
    [SerializeField] float speed;
    [SerializeField] float rotation;
    public Transform centerOverlabBox;
    Rigidbody rb;


    // Use this for initialization
    void Start () {
        rb = GetComponent<Rigidbody>();
    }

	  bool IsGrounded()
    {
        var listCols = Physics.OverlapSphere(centerOverlabBox.position, 0.01f);

        if (listCols.Length > 0) return true;
        else return false;
    }

	// Update is called once per frame
	void FixedUpdate () {

        var x = transform.position.x;
        var z = transform.position.z;
        var y = transform.position.y;

        if (IsGrounded())
        {
            z = transform.position.z + speed;

            playerTransform.position = new Vector3(x, y, z);

        }
        else
        {

            //this.transform.Rotate(0, 0, 10);
        }
        
    }


    void Awake()
    {
        playerTransform = transform;
    }
  
}
